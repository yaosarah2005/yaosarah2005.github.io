1.1 Altham’s Multiplicative Binomial Distribution

The PMF for Altham’s Multiplicative Binomial Distribution is \[f_X(x) = \frac{1}{c} {n \choose x} p^x (1-p)^{n-x} \theta^{x(n-x)}I(x \in \{0, 1, ..., n\})\] where \[c = \sum_{i=0}^n {n \choose i} p^i (1-p)^{n-i} \theta^{i(n-i)}.\] Just so we start on the same page, I have included my R function for computing probability using this distribution.

Code 
daltbinom <- function(x, size, prob, theta){
  normalizing.constant <- 0
  for(j in 0:size){
    normalizing.constant <- normalizing.constant + 
      choose(size, j) * prob^j * (1-prob)^(size-j) * theta^(j*(size-j))
  }
  # Probability Mass at x=x
  (1/normalizing.constant) *
    choose(size, x) * prob^x * (1-prob)^(size-x) * theta^(x*(size-x)) *
      (x>=0)*(x<=size)
}

2 Altham’s Multiplicative Beta Binomial Distribution

The PMF for Altham’s Multiplicative Beta Binomial Distribution is \[f_X(x) = \left[\int_{0}^1 \left(\frac{1}{c} {n \choose x} p^x (1-p)^{n-x} \theta^{x(n-x)}\right)\left(\frac{\Gamma(\alpha+\beta)}{\Gamma(\alpha)\Gamma(\beta)} p^{\alpha-1}(1-p)^{\beta-1}\right) dp\right]I(x \in \{0, 1, ..., n\})\]

Just so we start on the same page, I have included my R function for computing probability using this distribution.

Code 
daltbbinom.single <- function(x, size, alpha, beta, theta){
    integrand <- function(p) {
    sapply(p, # integrate will pass in a vector, so we use sapply to apply the following
           function(p_val){
             # f(x|p) * f(p|alpha, beta)
             daltbinom(x, size, p_val, theta) * dbeta(p_val, alpha, beta)
           }
    )
  }
  # Probability Mass at x=x
  # integrate(integrand, lower = 0, upper = 1)$value * (x>=0)*(x<=size)
  # I shrink the bounds below to avoid log(0)=-infinty
  # I use try() to catch when there is an error and avoid crashing
  res <- try(integrate(integrand, lower = 1e-7, upper = 1 - 1e-7)$value * 
                  (x>=0)*(x<=size), 
             silent = TRUE)
  # If there is an error, return a near-zero probability
  if(inherits(res, "try-error")) return(0.1^6)
  
  res
}
# Write a function that works on a vector
daltbbinom <- function(x, size, alpha, beta, theta){
  sapply(X=x, FUN=daltbbinom.single, size=size, alpha=alpha, beta=beta, theta=theta)
}

Code 
momdaltbinom <- function(p, theta, k){
  x <- 0:11
  sum(x^k*daltbinom(x,11,p,theta))
}

lldaltbinom <- function(par, data, MLE = F){
  p <- ifelse(MLE, 1/(1+exp(-par[1])), par[1])
  theta <- ifelse(MLE, exp(par[2]), par[2])
  ll <- sum(log(daltbinom(data,11,p,theta)))
  ifelse(MLE, -ll, ll)
}

player.turns <- table(c(dat.pdc.clean$Winner, dat.pdc.clean$Loser)) 

# This function returns a data frame that contains two column. 
# First column is number of legs the player wins in each match, 
# and the second column is the total legs that the player played for each match.
Win.each <- function(name){                      
  total.turn <- as.numeric(player.turns[name])
  win.each <- tibble(Win = rep(NA, total.turn),
                    Total = rep(NA, total.turn))
  n <- 1
  for(i in 1:nrow(dat.pdc.clean)){
    if(dat.pdc.clean$Winner[i]==name){ # If the player is the winner
      win.each$Win[n] <- 6
      win.each$Total[n] <- dat.pdc.clean$LoserLegs[i] + dat.pdc.clean$WinnerLegs[i]
      n <- n + 1
    }
    else if(dat.pdc.clean$Loser[i]==name){ # If the player is the loser
      win.each$Win[n] <- dat.pdc.clean$LoserLegs[i]
      win.each$Total[n] <- dat.pdc.clean$LoserLegs[i] + dat.pdc.clean$WinnerLegs[i]
      n <- n + 1 
    }
  }
  win.each
}

# This function calculates the probability of the player 
# winning x time in n legs with given probability and theta. 
# If the player win the match, it ensures the last leg is a win, 
# If the player lose the match, it ensures the last leg is a lose.
# Since player did not play 11 legs, size is also a vector that contains different total legs each player played.

Prob.daltbinom <- function(x, size, prob, theta){  
  sapply(1:length(x), function(i){  # Ensure it works on a vector
    xi <- x[i]  
    si <- size[i]  
    if(xi == 6){  # If the player win the match
      # Calculate the probability and ensuring the last leg is win
      (choose(si-1,5)/choose(si, xi)) * daltbinom(xi, si, prob, theta) 
    } else {  # If the player lose the match
      # Calculate the probability and ensuring the last leg is lose
      (choose(si-1, xi)/choose(si, xi)) * daltbinom(xi, si, prob, theta)  
    }  
  })  
}

# This function calculates the log likelihood of different parameters. 
# In order to make it easier for optimization, it returns the negative log likelihood.
llProb.daltbinom <- function(par, data){
  p <- 1/(1+exp(-par[1]))  # Ensures probability is between 0 and 1
  theta <- exp(par[2])     # Ensures theta is greater than 0
  x <- data$Win
  size <- data$Total
  ll <- sum(log(Prob.daltbinom(x,size,p,theta)))
  -ll
}

# Create a summary table for the result
MLE.summary <- tibble(Name = names(player.turns),
                      P = NA,
                      Theta = NA)

# Repeat the process for each player
for(i in 1:nrow(MLE.summary)){
  name <- MLE.summary$Name[i]
  win.each <- Win.each(name) 
  mle <- optim(par = c(0,0), fn = llProb.daltbinom, 
             data = win.each)
  MLE.summary$P[i] <- 1/(1+exp(-mle$par[1]))
  MLE.summary$Theta[i] <- exp(mle$par[2])
}

# This function calculates the probability of the player 
# wins x time with given alpha, beta, and theta. 
Prob.daltbbinom <- function(x, size, alpha, beta, theta){  
  sapply(1:length(x), function(i){  # Ensure it works on a vector
    xi <- x[i]  
    si <- size[i]  
    if(xi == 6){  # If the player win the match
      # Calculate the probability and ensuring the last leg is win
      (choose(si-1,5)/choose(si, xi)) * daltbbinom(xi, si, alpha, beta, theta) 
    } else {  # If the player lose the match
      # Calculate the probability and ensuring the last leg is lose
      (choose(si-1, xi)/choose(si, xi)) * daltbbinom(xi, si, alpha, beta, theta)  
    }  
  })  
}



# This function calculates the log likelihood of different parameters. 
# In order to make it easier for optimization, it returns the negative log likelihood.
llProb.daltbbinom <- function(par, data){
  alpha <- exp(par[1])
  beta <- exp(par[2]) 
  theta <- exp(par[3])    
  x <- data$Win
  size <- data$Total
  ll <- sum(log(Prob.daltbbinom(x,size,alpha, beta,theta)))
  -ll
}

  MLE.summary$Mean[i] <- alpha / (alpha + beta)

}
